 teams:any[] = [];
constructor(private svc: InfoPaginaService,private db:AngularFireDatabase) {
   db.list('/team').snapshotChanges().subscribe(res => {
      res.forEach(t => {
        const team = t.payload.toJSON(); console.log(t.payload.toJSON());

      });
    });

//////////////////////////////////////////////////////////////////////////////////////////////////////////FIREBASE

copiar todos los ASSETS del TEMPLATE  a la carpeta ASSETS que hay en SRC de angular
copiar todos los LINK tags del HEAD del file INDEX del TEMPLATE , al file INDEX de ANGULAR
copiar todos los SCRIPT tags del BODY del file INDEX del TEMPLATE , al file INDEX de ANGULAR
copiar la CLASS del BODY de HTML a ANGULAR
(hacer GIT COMMIT -M "1.0.0.X" despues de cada cambio importante. [Hacer ROLLBACK si es necesario usando "GIT CHECKOUT -- . " ])

1) HEADER
ng g c  shared/header --skip-tests -m app -d

copiar el HEADER del file HTML al COMPONENT "Header"


2) FOOTER
ng g c shared/footer -m app --skip-tests -d

copiar el FOOTER del file HTML al COMPONENT "Footer"


3) PAGES:
ng g c pages/portfolio  --skip-tests -m app -d
ng g c pages/about --skip-tests  -m app -d
ng g c pages/item --skip-tests -m app -d

copiar la SECTION principal (o el DIV principal) de cada page a su respectiva page

4) ROUTES:
Crear rutas en app-routing:
{ path: "", component: PortfolioComponent },
 ..........
{ path: "**", pathMatch: "full", redirectTo: "" }

agregar <router-outlet> 

5) MENU:
modificar LINKS en MENU a fin de usar ROUTES
usar: routerLink
